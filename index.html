package com.shoko.processor import android.content.Context import android.content.Intent import android.content.pm.PackageManager import android.net.Uri import android.provider.MediaStore import android.util.Log /** * CommandProcessor parses Gemini API responses and executes corresponding actions. * Supports ACTION_OPEN, ACTION_SEARCH, and ACTION_MEDIA commands. */ class CommandProcessor(private val context: Context) { companion object { private const val TAG = "CommandProcessor" private const val ACTION_OPEN = "ACTION_OPEN:" private const val ACTION_SEARCH = "ACTION_SEARCH:" private const val ACTION_MEDIA = "ACTION_MEDIA:" } // Map of common app names to their package names private val appNameToPackage = mapOf( "spotify" to "com.spotify.music", "youtube" to "com.google.android.youtube", "whatsapp" to "com.whatsapp", "instagram" to "com.instagram.android", "facebook" to "com.facebook.katana", "twitter" to "com.twitter.android", "gmail" to "com.google.android.gm", "maps" to "com.google.android.apps.maps", "chrome" to "com.android.chrome", "calculator" to "com.android.calculator2", "camera" to "com.android.camera", "gallery" to "com.google.android.apps.photos", "settings" to "com.android.settings", "messages" to "com.android.messaging", "phone" to "com.android.phone", "contacts" to "com.android.contacts", "clock" to "com.google.android.deskclock", "weather" to "com.google.android.apps.weather", "drive" to "com.google.android.apps.docs.editors.drive", "docs" to "com.google.android.apps.docs.editors.docs", "sheets" to "com.google.android.apps.docs.editors.sheets", "slides" to "com.google.android.apps.docs.editors.slides", "telegram" to "org.telegram.messenger", "discord" to "com.discord", "slack" to "com.Slack", "reddit" to "com.reddit.frontpage", "tiktok" to "com.zhiliaoapp.musically", "snapchat" to "com.snapchat.android", "viber" to "com.viber.voip", "line" to "jp.naver.line.android", "keep" to "com.google.android.keep" ) /** * Processes the response from Gemini API and executes the appropriate action. * @param response The response string from Gemini API * @return Boolean indicating if command was successfully processed */ fun processCommand(response: String): Boolean { return when { response.startsWith(ACTION_OPEN, ignoreCase = true) -> { handleOpenApp(response) } response.startsWith(ACTION_SEARCH, ignoreCase = true) -> { handleSearch(response) } response.startsWith(ACTION_MEDIA, ignoreCase = true) -> { handleMediaAction(response) } else -> { Log.d(TAG, "No recognized action in response: $response") false } } } /** * Handles ACTION_OPEN commands to launch applications. * Format: ACTION_OPEN:package_name or ACTION_OPEN:app_name */ private fun handleOpenApp(response: String): Boolean { return try { val appIdentifier = response.substringAfter(ACTION_OPEN, "").trim() if (appIdentifier.isEmpty()) { Log.e(TAG, "Empty app identifier in ACTION_OPEN") return false } // Resolve app name to package name if needed val packageName = if (appIdentifier.contains(".")) { // Assume it's already a package name appIdentifier } else { // Try to resolve common app name resolveAppName(appIdentifier) } if (packageName.isEmpty()) { Log.e(TAG, "Could not resolve app: $appIdentifier") return false } val packageManager = context.packageManager val launchIntent = packageManager.getLaunchIntentForPackage(packageName) if (launchIntent != null) { launchIntent.flags = Intent.FLAG_ACTIVITY_NEW_TASK context.startActivity(launchIntent) Log.d(TAG, "Launched app: $packageName") true } else { Log.e(TAG, "App not found: $packageName") false } } catch (e: Exception) { Log.e(TAG, "Error handling ACTION_OPEN", e) false } } /** * Handles ACTION_SEARCH commands to perform Google searches. * Format: ACTION_SEARCH:search_query */ private fun handleSearch(response: String): Boolean { return try { val query = response.substringAfter(ACTION_SEARCH, "").trim() if (query.isEmpty()) { Log.e(TAG, "Empty search query in ACTION_SEARCH") return false } val intent = Intent(Intent.ACTION_WEB_SEARCH).apply { putExtra(android.app.SearchManager.QUERY, query) flags = Intent.FLAG_ACTIVITY_NEW_TASK } context.startActivity(intent) Log.d(TAG, "Started search: $query") true } catch (e: Exception) { Log.e(TAG, "Error handling ACTION_SEARCH", e) false } } /** * Handles ACTION_MEDIA commands for media playback control. * Format: ACTION_MEDIA:play|pause|next|previous|stop */ private fun handleMediaAction(response: String): Boolean { return try { val action = response.substringAfter(ACTION_MEDIA, "").trim().lowercase() if (action.isEmpty()) { Log.e(TAG, "Empty media action in ACTION_MEDIA") return false } val mediaIntent = Intent(Intent.ACTION_MEDIA_BUTTON).apply { flags = Intent.FLAG_RECEIVER_FOREGROUND } val keycode = when (action) { "play" -> android.view.KeyEvent.KEYCODE_MEDIA_PLAY "pause" -> android.view.KeyEvent.KEYCODE_MEDIA_PAUSE "play_pause" -> android.view.KeyEvent.KEYCODE_MEDIA_PLAY_PAUSE "next" -> android.view.KeyEvent.KEYCODE_MEDIA_NEXT "previous" -> android.view.KeyEvent.KEYCODE_MEDIA_PREVIOUS "stop" -> android.view.KeyEvent.KEYCODE_MEDIA_STOP else -> { Log.e(TAG, "Unknown media action: $action") return false } } val keyEvent = android.view.KeyEvent( android.view.KeyEvent.ACTION_DOWN, keycode ) mediaIntent.putExtra(Intent.EXTRA_KEY_EVENT, keyEvent) context.sendBroadcast(mediaIntent) Log.d(TAG, "Sent media action: $action") true } catch (e: Exception) { Log.e(TAG, "Error handling ACTION_MEDIA", e) false } } /** * Resolves a common app name to its package name. * @param appName The human-readable app name (case-insensitive) * @return The package name, or empty string if not found */ private fun resolveAppName(appName: String): String { val lowerAppName = appName.lowercase().trim() return appNameToPackage[lowerAppName] ?: "" } /** * Gets the package name for a given app name. * Useful for debugging or external queries. * @param appName The human-readable app name * @return The package name, or null if not found */ fun getPackageName(appName: String): String? { return appNameToPackage[appName.lowercase()] } /** * Adds or updates an app mapping. * @param appName The human-readable app name * @param packageName The corresponding package name */ fun addAppMapping(appName: String, packageName: String) { (appNameToPackage as? MutableMap)?.put(appName.lowercase(), packageName) Log.d(TAG, "Added/updated mapping: $appName -> $packageName") } /** * Checks if an app is installed on the device. * @param packageName The package name to check * @return Boolean indicating if the app is installed */ fun isAppInstalled(packageName: String): Boolean { return try { context.packageManager.getApplicationInfo(packageName, 0) true } catch (e: PackageManager.NameNotFoundException) { false } } }
